pipeline {
    agent any
    stages {
        stage('Pre-Build') {
            steps {
                script {
                   bat 'mvn clean compile'
                }
            }
        }
       stage ('Build') {
          steps {
             script {
                bat 'mvn -B -DskipTests clean package'
            }
        }
    }
      stage('Test') {
            steps {
                bat 'mvn test'
            }
            post {
                always {
                    junit 'target/surefire-reports/*.xml'
                }
            }
        }  
}
}
